            <input type="text" id="userName" placeholder="اكتب اسمك" oninput="updateProfile()">
            <label>الحالة الصحية:</label>
            <select id="healthStatus" onchange="updateProfile()">
                <option value="none">سليم (لا توجد أمراض)</option>
                <option value="diabetes">مصاب بالسكري</option>
            </select>
            <label>الحساسية:</label>

    <script>
        const mealDB = {
            pizza: { name: "بيتزا خضار", cal: 280, ing: "دقيق، طماطم، جبن", sugar: "عالي", allergy: "جلوتين، ألبان" },
            burger: { name: "برجر مشوي", cal: 450, ing: "لحم، خبز، خس", sugar: "متوسط", allergy: "جلوتين" },
            salad: { name: "سلطة خضراء", cal: 110, ing: "خيار، طماطم، زيت", sugar: "منخفض", allergy: "لا يوجد" }
        };

        let userSettings = { name: "المستخدم", status: "none", allergy: "" };

        function switchPage(pageId, el) {
            document.querySelectorAll('.page').forEach(p => p.classList.remove('active-page'));
            document.getElementById(pageId).classList.add('active-page');
            document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
            el.classList.add('active');
        }

        function updateProfile() {
            userSettings.name = document.getElementById('userName').value || "المستخدم";
            userSettings.status = document.getElementById('healthStatus').value;
            userSettings.allergy = document.getElementById('allergyInput').value;
            document.getElementById('userNameDisplay').innerText = userSettings.name;
        }

        function handleCapture(event) {
            const preview = document.getElementById('capturePreview');
            preview.src = URL.createObjectURL(event.target.files[0]);
            preview.style.display = 'block';
            
            // Simulating AI recognition after picture
            setTimeout(() => {
                const mealName = prompt("ماذا أكلت؟ (بيتزا، برجر، سلطة)");
                if(mealName) showData('camera', mealName);
            }, 500);
        }

        function showData(mode, manualName = "") {
            const mealKey = mode === 'select' ? document.getElementById('mealSelect').value : manualName;
            const targetDiv = mode === 'select' ? document.getElementById('resultHome') : document.getElementById('resultCamera');
            
            const meal = mealDB[mealKey] || mealDB['pizza']; // Default to pizza if typed wrong

            let healthWarning = "";
            if (userSettings.status === 'diabetes' && meal.sugar === 'عالي') {
                healthWarning = `<div class="warning">⚠️ تنبيه: هذه الوجبة عالية السكر ولا تناسب حالتك (سكري).</div>`;
            }
            
            if (userSettings.allergy && meal.allergy.includes(userSettings.allergy)) {
                healthWarning += `<div class="warning">⚠️ تحذير: تحتوي الوجبة على (${meal.allergy}) وأنت تعاني من حساسية تجاهها!</div>`;
            }

            targetDiv.innerHTML = `
                <div class="card">
                    <h3>📊 تحليل الوجبة: ${meal.name}</h3>
                    <p>🔥 <strong>السعرات:</strong> ${meal.cal} سعرة</p>
                    <p>🥗 <strong>المكونات:</strong> ${meal.ing}</p>
                    ${healthWarning}
                    <div class="tip">💡 نصيحة: ${meal.cal > 300 ? 'حاول تقليل الكمية والمشي لمدة 20 دقيقة.' : 'وجبة رائعة وخفيفة، استمر!'}</div>
                </div>
            `;
        }
    </script>
